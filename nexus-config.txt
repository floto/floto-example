//goes in /etc/httpd/conf.d/nexus.conf
NameVirtualHost *:80

<VirtualHost *:80>
  ServerName localhost.local
  ProxyRequests Off

  <Proxy *>
    Order deny,allow
    Allow from all
  </Proxy>

  ProxyPass /nexus/ http://localhost:8081/nexus/
  ProxyPassReverse /nexus/ http://localhost:8081/nexus/
  ProxyPreserveHost On

  <Location />
    Order allow,deny
    Allow from all
  </Location>

  ErrorLog /var/www/sample-host/log/error.log
  CustomLog /var/www/sample-host/log/access.log common
</VirtualHost>

//goes in /home/user/www/nexus/conf
